Serializable dictionary

Serializable dictionary implementation to show/edit a C# dictionary in Unity inspector.



An error will show up when the dictionary cannot be desiarialized due to the fact that it contains repeating keys.

Show Code/Hide Code 
using System.Collections.Generic;
using UnityEngine;
#if UNITY_EDITOR
using UnityEditor;
#endif
 
[System.Serializable]
public class SerializableDictionary<TKey, TValue> : 
    Dictionary<TKey, TValue>, ISerializationCallbackReceiver
{
    [System.Serializable]
    public struct Pair<T1, T2>
    {
        public T1 key;
        public T2 value;
 
        public Pair(T1 key, T2 value)
        {
            this.key = key;
            this.value = value;
        }
    }
 
    [SerializeField] List<Pair<TKey, TValue>> _list = new List<Pair<TKey, TValue>>();
 
    [SerializeField] bool _isValid = true;
 
    // save the dictionary to lists
    public void OnBeforeSerialize()
    {
        if (!_isValid) return;
        _list.Clear();
        foreach (KeyValuePair<TKey, TValue> pair in this)
        {
            _list.Add(new Pair<TKey, TValue>(pair.Key, pair.Value));
        }
    }
 
    // load dictionary from lists
    public void OnAfterDeserialize()
    {
        _isValid = true;
        Clear();
        for (int i = 0; i < _list.Count; i++)
        {
            if (ContainsKey(_list[i].key))
            {
                _isValid = false;
                return;
            }
            Add(_list[i].key, _list[i].value);
        }
    }
}
 
#if UNITY_EDITOR
[CustomPropertyDrawer(typeof(SerializableDictionary<,>))]
public class SerializableDictionaryDrawer : PropertyDrawer
{
    public override void OnGUI(Rect position, SerializedProperty property, GUIContent label)
    {
        EditorGUI.BeginProperty(position, label, property);
        bool valid = property.FindPropertyRelative("_isValid").boolValue;
        if (!valid) GUI.color = Color.red;
        EditorGUI.PropertyField(position, property.FindPropertyRelative("_list"), label, true);
        if (!valid)
        {
            position.y += position.height - EditorGUIUtility.singleLineHeight;
            position.height = EditorGUIUtility.singleLineHeight;
            position.width = position.width - EditorGUIUtility.labelWidth;
            EditorGUI.HelpBox(position, "Repeating Keys", MessageType.Error);
            GUI.color = Color.white;
        }
        EditorGUI.EndProperty();
    }
 
    public override float GetPropertyHeight(SerializedProperty property, GUIContent label)
    {
        return EditorGUI.GetPropertyHeight(property.FindPropertyRelative("_list"));
    }
}

Mono Behaviour Ext

MonoBehaviour extension to add an unlimited amount of Start events created using C# reflection.

A class that inherits from MonoBehaviourExt cannot use Awake or Start , but can implement an unlimited amount of methods called Start0, Start1, Start2, etc...
These methods will be called automatically at the Start of each instance of classes that inherits from MonoBehaviourExt following the numeric order: First all the Start0 methods, then all the Start1 methods, etc…
Show Code/Hide Code 
using System;
using System.Collections.Generic;
using System.Reflection;
using UnityEngine;
 
public abstract class MonoBehaviourExt : MonoBehaviour
{
    static List<List<Action>> starts = new List<List<Action>>();
    static MonoBehaviourExt instance;
 
    public void Awake()
    {
        MethodInfo[] methodInfos = GetType().GetMethods(BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.Instance);
 
        foreach (MethodInfo methodInfo in methodInfos)
        {
            if (!methodInfo.Name.StartsWith("Start")) continue;
            try
            {
                int index = Int32.Parse(methodInfo.Name.Replace("Start", ""));
                for (int i = starts.Count; i <= index; i++) starts.Add(new List<Action>());
 
                Action action = (Action) methodInfo.CreateDelegate(typeof(Action), this);
                starts[index].Add(action);
            }
            catch (FormatException) { }
        }
        instance = this;
    }
 
    public void Start()
    {
        if (instance != this) return;
        List<List<Action>> tmpStarts = new List<List<Action>>(starts);
        starts.Clear();
        foreach (List<Action> actions in tmpStarts)
            foreach (Action action in actions)
                action.Invoke();
    }
}
